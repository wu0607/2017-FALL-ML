def area_to_class(y):
    if y==0:
        return 0
    elif y>0 and y<=1:
        return 1
    elif y>1 and y<=10:
        return 2
    elif y>10 and y<=100:
        return 3
    elif y>100 and y<=1000:
        return 4
    elif y>1000:
        return 5
